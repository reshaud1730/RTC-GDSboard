    };

    const sparkletRules = { green: { min: 5000, max: 30000 }, red: { min: 10000, max: 30000 }, yellow: { min: 15000, max: 30000 } };

    // Global Snapshots to handle re-renders properly
    window.pendingSnap = null;
    window.activeSnap = null;
    window.ledgerSnap = null;

    window.toggleAuth = () => {
        if (auth.currentUser) signOut(auth);
        else {
            const e = prompt("Admin Email:"), p = prompt("Password:");
            if(e && p) signInWithEmailAndPassword(auth, e, p).catch(() => alert("Access Denied"));
        }
    };

    onAuthStateChanged(auth, u => {
        document.querySelectorAll('.admin-only').forEach(el => el.style.display = u ? 'block' : 'none');
        document.getElementById('user-status').innerText = u ? "Mode: Admin Control" : "Mode: Fleet View";
        // Re-render UI now that Auth state has changed to show/hide delete buttons
        if(window.pendingSnap) renderPending(window.pendingSnap);
        if(window.activeSnap) renderActive(window.activeSnap);
        if(window.ledgerSnap) renderLedger(window.ledgerSnap);
    });

    window.submitForApproval = () => {
        const cat = document.getElementById('category-select').value;
        const addr = document.getElementById('project-address').value.trim();
        if(!cat || !addr) return alert("All fields required.");
        let name, value, color;
        if(cat === 'custom') {
            name = document.getElementById('custom-name').value;
            value = parseInt(document.getElementById('custom-hours').value);
            color = value < 40000 ? 'green' : value < 70000 ? 'red' : 'yellow';
        } else {
            const idx = document.getElementById('property-select').value;
            const p = properties[cat][idx];
            name = p.name; value = p.value;
            color = cat === 'service_structures' ? (value < 40000 ? 'green' : value < 70000 ? 'red' : 'yellow') : cat;
        }
        push(ref(db, 'pendingProjects'), { name, address: addr, value, color, maxGens: Math.ceil(value/10000) });
        alert("Build Submitted!");
        document.getElementById('project-address').value = ""; // Clear input
    };

    window.deleteItem = (folder, id) => {
        if(confirm("Permanently delete this item?")) {
            remove(ref(db, `${folder}/${id}`));
        }
    };

    function renderPending(snap) {
        const b = document.getElementById('pending-board');
        b.innerHTML = "";
        if(snap.val()) {
            Object.entries(snap.val()).forEach(([id, d]) => {
                const del = auth.currentUser ? `<button class="del-btn" onclick="window.deleteItem('pendingProjects','${id}')">🗑️ Delete</button>` : "";
                b.innerHTML += `<div class="project-card">${del}<strong>${d.name}</strong><br>${d.address}<br><button onclick="window.approve('${id}')" style="background:#2ecc71; padding:5px; font-size:12px; margin-top:10px;">Approve & Deploy</button></div>`;
            });
        } else {
            b.innerHTML = "<i>No pending approvals.</i>";
        }
    }

    onValue(ref(db, 'pendingProjects'), snap => {
        window.pendingSnap = snap;
        renderPending(snap);
    });

    window.approve = id => {
        onValue(ref(db, `pendingProjects/${id}`), s => {
            if(s.val()) { push(ref(db, 'activeProjects'), { ...s.val(), currentGens: 0 }); remove(ref(db, `pendingProjects/${id}`)); }
        }, { onlyOnce: true });
    };

    function renderActive(snap) {
        const b = document.getElementById('dispatch-board');
        b.innerHTML = "";
        window.activeProjectsArr = snap.val() ? Object.entries(snap.val()).map(([id, d]) => ({id, ...d})) : [];
        if(snap.val()) {
            Object.entries(snap.val()).forEach(([id, p]) => {
                const del = auth.currentUser ? `<button class="del-btn" onclick="window.deleteItem('activeProjects','${id}')">🗑️ Delete</button>` : "";
                b.innerHTML += `<div class="project-card ${p.color}-gen" style="min-width:250px;">${del}<strong>${p.name}</strong><br>${p.address}<br>Slots: ${p.maxGens - p.currentGens} Open<br><button onclick="window.accept('${id}')" style="margin-top:10px; padding:8px; font-size:14px;">Claim Slot</button></div>`;
            });
        } else {
            b.innerHTML = "<i>No active projects.</i>";
        }
    }

    onValue(ref(db, 'activeProjects'), snap => {
        window.activeSnap = snap;
        renderActive(snap);
    });

    window.accept = id => {
        const p = window.activeProjectsArr.find(x => x.id === id);
        const user = prompt("Upland Username:");
        if(!user) return;
        const gens = parseInt(prompt(`Gens? (Max ${p.maxGens - p.currentGens})`));
        const rules = sparkletRules[p.color];
        const sparklet = parseInt(prompt(`Sparklet per gen? (${rules.min}-${rules.max})`));
        if(!gens || !sparklet || sparklet < rules.min || sparklet > rules.max) return alert("Invalid entry.");
        const power = (gens * sparklet) / 1000;
        const hours = p.value / power;
        push(ref(db, 'ledgerLog'), { 
            name: p.name, address: p.address, subcontractor: user, activeSpark: power, 
            status: 'active', timestamp: Date.now(), hoursNeeded: hours, color: p.color,
            endTime: Date.now() + (hours * 3600000), sparkletPerGen: sparklet, genCount: gens
        });
        const nc = p.currentGens + gens;
        if(nc >= p.maxGens) remove(ref(db, `activeProjects/${id}`));
        else update(ref(db, `activeProjects/${id}`), { currentGens: nc });
    };

    function renderLedger(snap) {
        window.ledgerArr = snap.val() ? Object.entries(snap.val()).map(([id, d]) => ({id, ...d})) : [];
        const adminList = document.getElementById('active-stakes-admin');
        const weeklyBody = document.getElementById('weekly-body');
        const monthlyBody = document.getElementById('monthly-body');
        const leaderboardUI = document.getElementById('leaderboard-ui');
        const isAdmin = auth.currentUser;

        adminList.innerHTML = ""; weeklyBody.innerHTML = ""; monthlyBody.innerHTML = ""; leaderboardUI.innerHTML = "";

        let totalPower = 0;
        const now = Date.now();
        const userStats = {};

        window.ledgerArr.forEach(s => {
            userStats[s.subcontractor] = (userStats[s.subcontractor] || 0) + s.activeSpark;
            const delBtn = isAdmin ? `<button class="del-btn" onclick="window.deleteItem('ledgerLog','${s.id}')">🗑️ Delete</button>` : "";
            
            if(s.status === 'active') {
                totalPower += s.activeSpark;
                // Add "countdown-timer" class and "data-endtime" so setInterval can update text without destroying buttons
                adminList.innerHTML += `<div class="project-card">${delBtn}<strong>${s.name} (${s.subcontractor})</strong><div class="timer-display countdown-timer" data-endtime="${s.endTime}">Loading Timer...</div><button onclick="window.finalize('${s.id}')" style="background:#e67e22; padding:8px; font-size:14px; margin-top:10px;">Finalize & Pay Out</button></div>`;
            } else {
                const date = new Date(s.timestamp).toLocaleDateString();
                const row = `<tr><td>${s.subcontractor}</td><td>${s.name}</td><td>${s.payout}</td><td>${date}</td><td class="admin-only">${delBtn}</td></tr>`;
                if(s.timestamp > now - 604800000) weeklyBody.innerHTML += row;
                if(s.timestamp > now - 2592000000) monthlyBody.innerHTML += row;
            }
        });

        const topFleet = Object.entries(userStats).sort((a,b) => b[1] - a[1]).slice(0, 5);
        topFleet.forEach(([user, power], i) => {
            leaderboardUI.innerHTML += `<div class="leaderboard-item"><span>#${i+1} ${user}</span><span>${power.toFixed(2)} Spark</span></div>`;
        });
        document.getElementById('total-spark-stat').innerText = totalPower.toFixed(2);
        document.getElementById('top-member').innerText = topFleet[0] ? topFleet[0][0] : "---";
        
        // Ensure Admin columns are visible if logged in
        document.querySelectorAll('.admin-only').forEach(el => el.style.display = isAdmin ? 'table-cell' : 'none');
        // Trigger a timer update immediately
        updateTimers();
    }

    onValue(ref(db, 'ledgerLog'), snap => {
        window.ledgerSnap = snap;
        renderLedger(snap);
    });

    window.finalize = (id) => {
        const s = window.ledgerArr.find(x => x.id === id);
        const h = Math.min(s.hoursNeeded, (Date.now() - s.timestamp) / 3600000);
        const p = Math.floor((s.activeSpark * h) * 0.95);
        window.generateInvoice(s, p);
        update(ref(db, `ledgerLog/${id}`), { status: 'finished', payout: p, timestamp: Date.now() });
    };

    window.generateInvoice = (data, finalPayout) => {
        const { jsPDF } = window.jspdf;
        const doc = new jsPDF();
        doc.setFontSize(22);
        doc.text("RAYTOWN CONSTRUCTION", 20, 20);
        doc.setFontSize(12);
        doc.text("Project Payout Receipt", 20, 30);
        doc.line(20, 35, 190, 35);
        doc.text(`Subcontractor: ${data.subcontractor}`, 20, 50);
        doc.text(`Property: ${data.name}`, 20, 60);
        doc.text(`Address: ${data.address}`, 20, 70);
        doc.text(`Generators Deployed: ${data.genCount}`, 20, 90);
        doc.text(`Sparklet per Gen: ${data.sparkletPerGen}`, 20, 100);
        doc.text(`Total Spark Power: ${data.activeSpark.toFixed(2)}`, 20, 110);
        doc.setFontSize(16);
        doc.text(`TOTAL PAYOUT (95%): ${finalPayout.toLocaleString()} UPX`, 20, 130);
        doc.setFontSize(10);
        doc.text("Thank you for building with the Raytown Fleet.", 20, 150);
        doc.text(`Issued: ${new Date().toLocaleString()}`, 20, 160);
        doc.save(`${data.subcontractor}_Invoice.pdf`);
    };

    function formatTime(ms) {
        const h = Math.floor(ms / 3600000);
        const m = Math.floor((ms % 3600000) / 60000);
        const s = Math.floor((ms % 60000) / 1000);
        return `${h}h ${m}m ${s}s`;
    }

    window.populateProperties = () => {
        const cat = document.getElementById('category-select').value;
        const sel = document.getElementById('property-select');
        const cG = document.getElementById('custom-group');
        const pG = document.getElementById('property-group');
        cG.style.display = cat === 'custom' ? 'block' : 'none';
        pG.style.display = cat === 'custom' ? 'none' : 'block';
        sel.innerHTML = "";
        if(cat && properties[cat]) {
            sel.disabled = false;
            properties[cat].forEach((p, i) => sel.innerHTML += `<option value="${i}">${p.name} (${p.value.toLocaleString()} Hrs)</option>`);
        } else sel.disabled = true;
    };

    // FIX: Update timers WITHOUT re-rendering the buttons so clicks aren't stolen!
    function updateTimers() {
        const now = Date.now();
        document.querySelectorAll('.countdown-timer').forEach(el => {
            const end = parseInt(el.getAttribute('data-endtime'));
            const rem = end - now;
            el.innerText = rem > 0 ? formatTime(rem) : "DONE - READY FOR PAYOUT";
        });
    }

    setInterval(() => { 
        document.getElementById('live-clock').innerText = new Date().toLocaleString(); 
        updateTimers(); 
    }, 1000);
